30  * Saturated addition where the RHS is assumed to be a positive value.
31  *
32  * This is part of the generic, promotionless, saturation math provided by this header. Ideally this would
33  * be omitted for most platforms where saturation operations were available.
34  */
35 template <
36  class SIGNED_TYPE,
37  typename std::enable_if<std::is_integral<SIGNED_TYPE>::value && std::is_signed<SIGNED_TYPE>::value, int>::type = 0>
38 constexpr SIGNED_TYPE _saturating_add_d(SIGNED_TYPE left, SIGNED_TYPE right)
39 {
40  return (left >= 0 && right > (std::numeric_limits<SIGNED_TYPE>::max() - left))
41  ? std::numeric_limits<SIGNED_TYPE>::max()
42  : static_cast<SIGNED_TYPE>(left + right);
43 }

45 /*
46  * Saturated subtraction where the RHS is assumed to be a positive value.
47  *
48  * This is part of the generic, promotionless, saturation math provided by this header. Ideally this would
49  * be omitted for most platforms where saturation operations were available.
50  */
51 template <
52  class SIGNED_TYPE,
53  typename std::enable_if<std::is_integral<SIGNED_TYPE>::value && std::is_signed<SIGNED_TYPE>::value, int>::type = 0>
54 constexpr SIGNED_TYPE _saturating_sub_d(SIGNED_TYPE left, SIGNED_TYPE right)
55 {

73 template <
74  class SIGNED_TYPE,
75  typename std::enable_if<std::is_integral<SIGNED_TYPE>::value && std::is_signed<SIGNED_TYPE>::value, int>::type = 0>
76 SIGNED_TYPE saturating_add(SIGNED_TYPE left, SIGNED_TYPE right)
77 {
78  SIGNED_TYPE result;
79  if (right >= 0) // make sure this is actually substraction
80  {
81  result = _saturating_add_d<SIGNED_TYPE>(left, right);
82  }
83  else if (right == std::numeric_limits<SIGNED_TYPE>::min())
84  {
85  static_assert(std::numeric_limits<SIGNED_TYPE>::max() + std::numeric_limits<SIGNED_TYPE>::min() == -1,
86  "This header assumes two's complement integer representations.");
87  // For two's compliment the abs(max) !- abs(min). This is the only
88  // place we assume two's compliment. If we need to port this to another integer representation
89  // this should be the only thing we need to fix.
90  result = _saturating_sub_d<SIGNED_TYPE>(left, std::numeric_limits<SIGNED_TYPE>::max());
91  if (result > std::numeric_limits<SIGNED_TYPE>::min())
92  {
93  result = static_cast<SIGNED_TYPE>(result - 1);
94  }
95  }
96  else
97  {
98  // Flip the sign-bit and delegate to the subtraction implementation.
99  // We assume that the integer representation can flip the bit without
100  // promotion where right > std::numeric_limits<SIGNED_TYPE>::min().
101  result = _saturating_sub_d<SIGNED_TYPE>(left, static_cast<SIGNED_TYPE>(-right));
102  }
103  return result;
104 }
